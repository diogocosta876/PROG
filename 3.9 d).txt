bool IsLeap(int year) {
	if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0)
		return true;
	else
		return false;
}

string dayOfWeek(int day, int month, int year) {
	int s;     //two first digits year
	int a;     //two last digits year
	int c;        //monthcode
	int dayweek_number;
	string dayweek;

	s = year / 100; //outputs int
	a = year % 100;

	c = monthCode(month, IsLeap(month));

	dayweek_number = ((5 * a / 4) + c + day - 2 * (s % 4) + 7) % 7;

	switch (dayweek_number) {
	case 0:
		dayweek = "Saturday";
		break;
	case 1:
		dayweek = "Sunday";
		break;
	case 2:
		dayweek = "Monday";
		break;
	case 3:
		dayweek = "Tuesday";
		break;
	case 4:
		dayweek = "Wednesday";
		break;
	case 5:
		dayweek = "Thursday";
		break;
	case 6:
		dayweek = "Friday";
		break;
	}

	return dayweek;
}

void displayMonth(int month, int year) {
	vector<int> sun;
	vector<int> mon;
	vector<int> tue;
	vector<int> wed;
	vector<int> thu;
	vector<int> fri;
	vector<int> sat;

	for (int i = 1; i <= daysInMonth(month, year); i++) {

		string day = dayOfWeek(i, month, year);
		if (day == "Saturday")
			sat.push_back(i);
		if (day == "Sunday")
			sun.push_back(i);
		if (day == "Monday")
			mon.push_back(i);
		if (day == "Tuesday")
			tue.push_back(i);
		if (day == "Wednesday")
			wed.push_back(i);
		if (day == "Thursday")
			thu.push_back(i);
		if (day == "Friday")
			fri.push_back(i);
	}

	int i = 0;
	while (i < daysInMonth(month, year)) {
		if (find(sun.begin(), sun.end(), i) != sun.end()) {
			cout << i << "\t";
			i++;
		}
		else {
			cout << " \t";
		}
		if (find(mon.begin(), mon.end(), i) != mon.end()) {
			cout << i << "\t";
			i++;
		}
		else
			cout << " \t";
		if (find(tue.begin(), tue.end(), i) != tue.end()) {
			cout << i << "\t";
			i++;
		}
		else
			cout << " \t";
		if (find(wed.begin(), wed.end(), i) != wed.end()) {
			cout << i << "\t";
			i++;
		}
		else
			cout << " \t";
		i++;
		if (find(thu.begin(), thu.end(), i) != thu.end()) {
			cout << i << "\t";
			i++;
		}
		else
			cout << " \t";
		if (find(fri.begin(), fri.end(), i) != fri.end()) {
			cout << i << "\t";
			i++;
		}
		else
			cout << " \t";
		if (find(sat.begin(), sat.end(), i) != sat.end()) {
			cout << i << "\n";
			i++;
		}
		else
			cout << " \n";
	}
	cout << "\n" << endl;
}
