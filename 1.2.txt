#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
	int a, b, c;
	double mean;
	cout << "Please, input 3 integer numbers\n";
	cout << "A ? "; cin >> a;
	cout << "B ? "; cin >> b;
	cout << "C ? "; cin >> c;
	mean = (a + b + c) / 3.0;
	cout << "mean = " << mean << "\n";
	cout << "A-mean = " << a - mean << "\n";
	cout << "B-mean = " << b - mean << "\n";
	cout << "C-mean = " << c - mean << "\n";
	return 0;
}
