#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
    long long a, b;
    cout << "input 2 integers: "; cin >> a;
    cin >> b;

    if (a + b > INT_MAX) {
        cout << "sum overflow" << endl;
    }
    else if (a + b < INT_MIN) {
        cout << "sum underflow" << endl;
    }
    else {
        cout << "possible" << endl;
        cout << a + b;
    }

    return 0;
}
