bool IsLeap(int year) {
	if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0)
		return true;
	else
		return false;
}

string dayOfWeek(int day, int month, int year) {
	int s;     //two first digits year
	int a;     //two last digits year
	int c;        //monthcode
	int dayweek_number;
	string dayweek;

	s = year / 100; //outputs int
	a = year % 100;

	c = monthCode(month, IsLeap(month));

	dayweek_number = ((5 * a / 4) + c + day - 2 * (s % 4) + 7) % 7;
	
	switch (dayweek_number) {
	case 0:
		dayweek = "Saturday";
		break;
	case 1:
		dayweek = "Sunday";
		break;
	case 2:
		dayweek = "Monday";
		break;
	case 3:
		dayweek = "Tuesday";
		break;
	case 4:
		dayweek = "Wednesday";
		break;
	case 5:
		dayweek = "Thursday";
		break;
	case 6:
		dayweek = "Friday";
		break;
	}

	return dayweek;
}
