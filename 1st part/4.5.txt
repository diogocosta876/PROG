#include <iostream>
#include <iomanip>
#include <string>
#include <algorithm>
#include <vector>
#include <climits>
#include <cmath>
using namespace std;

void printvector(vector<int> s) {
	for (int i = 0; i < s.size(); i++) {
		cout << s[i] << " ";
	}
	cout << endl;
}

void removeDuplicates(vector<int>& s) {
	for (int i = 0; i < s.size() - 1; i++) {      //initialize each number's duplicate checking
		for (int j = i + 1; j < s.size(); j++) {  //loop though every number to find duplicates
			if (s[i] == s[j]) {
				int k = j;
				while (k < s.size() - 1) {   // once found, push every number back 1 position
					s[k] = s[k + 1];
					k++;
				}
				s.resize(s.size() - 1);
				printvector(s); cout << "\n";
				j--;
			}
		}
	}
}


int main() {
	vector<int> s = { 1, 2, 3, 1, 1 };
	printvector(s); cout << "\n";
	removeDuplicates(s);
	printvector(s); cout << "\n";
	return 0;
}
