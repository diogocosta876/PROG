#include <iostream>
#include <iomanip>
#include <cmath>
#include <vector>
using namespace std;

int main() {
    
    vector <int> list;
    int n;
    cout << "input n: "; cin >> n;

    while (n != 1) {
        for (int i = 2; i <= n; i++) {
            if (n % i == 0) {
                n = n / i;
                list.push_back(i);
                break;
            }
        }
    }
    // OUTPUT VECTOR LIST
    int length = list.size();
    for (int j = 0; j < length; j++) {
        if (j < (length - 1)) {
            cout << list[j] << "x";
        }
        else {
            cout << list[j];
        }
    }
    return 0;
}
