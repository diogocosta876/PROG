#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
    int n;
    bool prime = true;
    cout << "input n: "; cin >> n;
    
    if (n == 0 or n == 1) {
        cout << "not prime";
        prime = false;
    }
    else {
        for (int i = 2; i < sqrt(n) + 1; i++) {
            if (n % i == 0) {
                cout << "not prime";
                prime = false;
                break;
            }
        }
    }
    if (prime) {
        cout << "prime";
    }
    return 0;
}
