#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
    unsigned long n = 0 - 1;  //biggest unsigned long

    for (n; n > 1; n--) {
        bool prime = true;

        if (n == 0 or n == 1) {
            prime = false;
        }
        else {
            for (int i = 2; i < sqrt(n) + 1; i++) {
                if (n % i == 0) {
                    prime = false;
                    break;
                }
            }
        }
        if (prime) {
            cout << n << " ";
            break;
        }
    }
    return 0;
}
