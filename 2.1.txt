#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
	double a, b, c, d, e, f;
	double x, y;

	cout << "a, b, c, d, e, f \n";
	cin >> a >> b >> c >> d >> e >> f;

	if (a * e - b * d == 0) {
		cout << "impossible system" << endl;
	}
	else {
		x = (c * e - b * f) / (a * e - b * d);
		y = (a * f - c * d) / (a * e - b * d);

		cout << "x, y: " << x << " " << y;
	}
	return 0;
}
