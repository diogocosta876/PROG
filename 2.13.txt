#include <iostream>
#include <iomanip>
#include <cmath>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    
    vector <int> list;
    int n, sum = 0; double mean;


    while (true) {
        cout << "input integer: "; cin >> n;
        if (cin.eof() == 0) {
            list.push_back(n);
            sum += n;
        }
        else {
            break;
        }
    }

    cout << "sequence: ";
    for (int i = 0; i < list.size(); i++) {
        cout << list.at(i) << " ";
        if (i == list.size() - 1) {
            cout << "\n";
        }
    }

    cout << "sum: " << sum << endl;
    cout << "mean: " << sum / list.size() << endl;

    sort(list.begin(), list.end());
    int min = list[0];
    int last = list.size() - 1;
    int max = list[last];
    cout << "Min value: " << min << endl;
    cout << "Max value: " << max << endl;

    return 0;
}
